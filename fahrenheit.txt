// Função para converter Celsius para Fahrenheit
fun celsiusParaFahrenheit(celsius) {
  return (celsius * 9 / 5) + 32;
}

print "--- Conversor de Temperatura ---";
var tempCelsius = 25;
var tempFahrenheit = celsiusParaFahrenheit(tempCelsius);
// Correção: Concatene com strings vazias para garantir que tudo seja string antes do '+'
print "" + tempCelsius + "°C é igual a " + tempFahrenheit + "°F."; // Deve imprimir 77°F

tempCelsius = 0;
tempFahrenheit = celsiusParaFahrenheit(tempCelsius);
print "" + tempCelsius + "°C é igual a " + tempFahrenheit + "°F."; // Deve imprimir 32°F

tempCelsius = -10;
tempFahrenheit = celsiusParaFahrenheit(tempCelsius);
print "" + tempCelsius + "°C é igual a " + tempFahrenheit + "°F."; // Deve imprimir 14°F
